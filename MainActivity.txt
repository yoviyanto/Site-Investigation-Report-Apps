public class MainActivity extends AppCompatActivity {
    private Button button;
    EditText mSiteIdData, mSiteNameData, mLongitudeData, mLatitudeData, mTowerHighData;
    String stringSiteIdData, stringSiteNameData, stringLongitudeData, stringLatitudeData, stringTowerHighData;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        button = (Button) findViewById(R.id.button);
        button.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                openActivityAntenna();
            }
        });
    }
    public void openActivityAntenna() {
        Intent intent = new Intent(MainActivity.this, ActivityAntenna.class);

        mSiteIdData = (EditText) findViewById(R.id.siteIdData);
        stringSiteIdData =  mSiteIdData.getText().toString();

        mSiteNameData = (EditText) findViewById(R.id.siteNameData);
        stringSiteNameData =  mSiteNameData.getText().toString();

        mLongitudeData = (EditText) findViewById(R.id.longitudeData);
        stringLongitudeData =  mLongitudeData.getText().toString();

        mLatitudeData = (EditText) findViewById(R.id.latitudeData);
        stringLatitudeData =  mLatitudeData.getText().toString();

        mTowerHighData = (EditText) findViewById(R.id.towerHighData);
        stringTowerHighData =  mTowerHighData.getText().toString();


        intent.putExtra("xtraSiteIdData", stringSiteIdData);
        intent.putExtra("xtraSiteNameData", stringSiteNameData);
        intent.putExtra("xtraLongitudeData", stringLongitudeData);
        intent.putExtra("xtraLatitudeData", stringLatitudeData);
        intent.putExtra("xtraTowerHighData", stringTowerHighData);

        startActivity(intent);

    }

}
